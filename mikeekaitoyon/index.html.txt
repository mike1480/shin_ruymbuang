  <script>
    (function() {
      // ----- flower data -----
      const flowerMap = {
        rose: { emoji: '🌹', message: 'i love u · you\'re cute 🌹', galleryEmoji: '🌹' },
        tulip: { emoji: '🌷', message: 'tulips for my tulip, you\'re so sweet 🌷💕', galleryEmoji: '🌷' },
        daisy: { emoji: '🌼', message: 'daisies and daydreams of you 🌼💖', galleryEmoji: '🌼' },
        sunflower: { emoji: '🌻', message: 'you\'re my sunshine, my only sunflower 🌻😘', galleryEmoji: '🌻' },
        cherry: { emoji: '🌸', message: 'cherry blossom love, soft and true 🌸✨', galleryEmoji: '🌸' }
      };

      // DOM elements
      const heartEl = document.getElementById('heartButton');
      const mainFlowerEmoji = document.getElementById('mainFlowerEmoji');
      const mainLoveMessage = document.getElementById('mainLoveMessage');
      const flowerGallery = document.getElementById('flowerGallery');
      const complimentPopup = document.getElementById('complimentPopup');

      // all flower buttons
      const flowerButtons = document.querySelectorAll('.flower-btn');

      // compliments array for heart click
      const compliments = [
        '💋 i love you!', 
        'you\'re cute 💕', 
        'my heart blooms 🌸', 
        'you make me smile 😘', 
        'endless love 💖', 
        'you\'re adorable ✨', 
        'love u to the moon 🌙❤️', 
        'sweet as a rose 🌹'
      ];

      // ---- helper: update flower display ----
      function setActiveFlower(flowerKey, customMessage = null) {
        const flower = flowerMap[flowerKey];
        if (!flower) return;

        // change main emoji
        mainFlowerEmoji.textContent = flower.emoji;

        // change message (if customMessage from button, else default)
        if (customMessage) {
          mainLoveMessage.textContent = customMessage;
        } else {
          mainLoveMessage.textContent = flower.message;
        }

        // update gallery: fill with 7 of the same flower emoji (more flowers!)
        const galleryEmoji = flower.galleryEmoji;
        let galleryHtml = '';
        for (let i = 0; i < 8; i++) {  // 8 flowers in gallery
          galleryHtml += <span class="garden-emoji">${galleryEmoji}</span>;
        }
        flowerGallery.innerHTML = galleryHtml;

        // add blooming animation to main flower
        mainFlowerEmoji.style.animation = 'none';
        void mainFlowerEmoji.offsetWidth; // reflow trick
        mainFlowerEmoji.style.animation = 'bloomGlow 2s infinite alternate, gentleSpin 12s infinite linear';
      }

      // ---- heart click: show random compliment near heart ----
      heartEl.addEventListener('click', function(e) {
        // pick random compliment
        const randomIndex = Math.floor(Math.random() * compliments.length);
        const msg = compliments[randomIndex];
        complimentPopup.textContent = msg;

        // position popup near heart
        const rect = heartEl.getBoundingClientRect();
        const glassRect = document.querySelector('.glass-card').getBoundingClientRect();

        // calculate relative position inside .glass-card
        const popupLeft = rect.left - glassRect.left + (rect.width / 2) - 50; // approx center
        const popupTop = rect.top - glassRect.top - 45; // above heart

        complimentPopup.style.left = popupLeft + 'px';
        complimentPopup.style.top = popupTop + 'px';
        complimentPopup.style.opacity = '1';
        complimentPopup.style.transform = 'scale(1.1)';

        // hide after 1.7 seconds
        clearTimeout(window.complimentTimeout);
        window.complimentTimeout = setTimeout(() => {
          complimentPopup.style.opacity = '0';
        }, 1700);

        // add micro heart animation on click
        heartEl.style.transform = 'scale(1.2)';
        setTimeout(() => { heartEl.style.transform = ''; }, 200);
      });

      // ---- flower button click handlers ----
      flowerButtons.forEach(btn => {
        btn.addEventListener('click', function() {
          const flowerType = this.dataset.flower;     // rose, tulip, etc.
          const customMsg = this.dataset.message;     // specific message from button
          setActiveFlower(flowerType, customMsg);

          // also make the flower jump a little
          mainFlowerEmoji.style.transform = 'scale(1.2)';
          setTimeout(() => mainFlowerEmoji.style.transform = '', 200);

          // extra: change background glass tint subtle (just for fun)
          const glass = document.querySelector('.glass-card');
          glass.style.transition = 'background 0.4s';
          if (flowerType === 'rose') glass.style.background = 'rgba(255, 230, 245, 0.7)';
          else if (flowerType === 'tulip') glass.style.background = 'rgba(235, 225, 255, 0.7)';
          else if (flowerType === 'daisy') glass.style.background = 'rgba(255, 250, 220, 0.7)';
          else if (flowerType === 'sunflower') glass.style.background = 'rgba(255, 240, 200, 0.75)';
          else if (flowerType === 'cherry') glass.style.background = 'rgba(255, 225, 240, 0.75)';
          setTimeout(() => {
            glass.style.background = 'rgba(255, 245, 250, 0.65)'; // revert after 0.5s
          }, 500);
        });
      });

      // ---- set default flower to rose on page load ----
      setActiveFlower('rose', 'i love u · you\'re cute 🌹');

      // ---- also you can click on the giant flower for a tiny surprise 🙂 ----
      mainFlowerEmoji.addEventListener('click', function(e) {
        // show mini message near flower
        complimentPopup.textContent = '💞 you\'re the best!';
        const rect = mainFlowerEmoji.getBoundingClientRect();
        const glassRect = document.querySelector('.glass-card').getBoundingClientRect();
        complimentPopup.style.left = (rect.left - glassRect.left + 20) + 'px';
        complimentPopup.style.top = (rect.top - glassRect.top - 30) + 'px';
        complimentPopup.style.opacity = '1';
        setTimeout(() => { complimentPopup.style.opacity = '0'; }, 1400);

        // bounce animation
        mainFlowerEmoji.style.transform = 'rotate(5deg) scale(1.1)';
        setTimeout(() => mainFlowerEmoji.style.transform = '', 300);
      });

      // ---- extra blooming detail: floating hearts appear near cursor? no, but we have plenty.
      // make the message extra romantic by letting user click anywhere on message?
      mainLoveMessage.addEventListener('click', function() {
        mainLoveMessage.style.transform = 'scale(1.05)';
        mainLoveMessage.style.background = '#ffe8f0';
        setTimeout(() => {
          mainLoveMessage.style.transform = '';
          mainLoveMessage.style.background = ''; 
        }, 300);
      });

    })();
  </script>
